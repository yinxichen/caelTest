                    $(function () {
                        var data = [{ id: 1, text: "南京" }, { id: 2, text: "北京" }, { id: 3, text: "西安" },
                        { id: 4, text: "重庆" }, { id: 5, text: "广州" }];
                        // 单选本地数据
                        var select1 = $("#select1").jkj("easySelect", {
                            data: data
                        });
                        $('#select1-load').on('click', function(e){
                            e.preventDefault();
                            var newData=[];
                            $(data).each(function(index, obj){
                                var item = {};
                                item.id = obj.id;
                                item.text = $.jkj.util.random.getLongDateString();
                                newData.push(item);
                            });
                            select1.loadData(newData, true, null);
                        });
                    })
                </script>

    自动完成配置参数 
                
    调用方法: $.fn.jkj('easySelect',options)
    
            
    .clear() 清除选中的值
    
            
    .clearAll() 清除选中的值和下拉选项等
    
            
    .loadData(data, clear, defaultValue) 加载新的数据
    
                data：加载的数据
    
                clear：boolean类型，是否清除老数据
    
                defaultValue：默认值。undefined保持原来不变，null清空值，其他为字符串、数值、数组（多选）

                    $(function () {

                        var data = [{ id: 1, text: "南京" }, { id: 2, text: "北京" }, { id: 3, text: "西安" },
                        { id: 4, text: "重庆" }, { id: 5, text: "广州" }];

                        // 单选 local data
                        // tags启用，输入的内容可以动态添加到下拉列表中
                        var select1 = $.jkj.select2.local("#select1", {
                            data: data,
                            tags: true
                        });

                        // 单选 ajax load once
                        var select2 = $.jkj.select2.remote("#select2", {
                            ajax: {
                                url: "/src/data/select2.json",
                                extra: { name: "test" },
                                success: null
                            }
                        });

                        // 多选 local data
                        var select3 = $.jkj.select2.local("#select3", { data: data });

                        var select4 = $.jkj.select2.local("#select4", { data: data, disabled: true });

                        var select6 = $.jkj.select2.autocomplete("#select6", {
                            paramName: 'name',
                            initSelection: function (element, callback) {         // 初始化时设置默认值
                                callback({ id: 1, realName: '123', companyName: 'adfa' });
                            },
                            ajax: {
                                url: "/src/data/select6.json",
                            },
                            formatResult: "<span><font style=\"font-family:Microsoft YaHei;\">{realName}</font><br/><b><i>{companyName}</i></b></span>",
                            formatSelection: function () {
                                return "<b>{realName}<b>";
                            }
                        });
                        //移除事件
                        $('#select6').on('select2:unselect', function () {
                            alert('select2:unselect')
                        });

                        // 单选 ajax load once                   
                        var select8 = $.jkj.select2.autocomplete("#select8", {
                            ajax: {
                                url: "/AdminJKJ/src/data/select2.json",
                                extra: { name: "test" },
                                success: null
                            },

                        });
                        $('#select8').append(new Option('123456', 12, true, true)).trigger('change');
                    });
                </script>
