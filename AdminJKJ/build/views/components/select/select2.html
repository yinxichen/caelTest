                    $(function () {
                        var data = [{ id: 1, text: "南京" }, { id: 2, text: "北京" }, { id: 3, text: "西安" },
                        { id: 4, text: "重庆" }, { id: 5, text: "广州" }];
                        // 单选本地数据
                        var select1 = $("#select1").jkj("easySelect", {
                            data: data
                        });
                        $('#select1-load').on('click', function(e){
                            e.preventDefault();
                            var newData=[];
                            $(data).each(function(index, obj){
                                var item = {};
                                item.id = obj.id;
                                item.text = $.jkj.util.random.getLongDateString();
                                newData.push(item);
                            });
                            select1.loadData(newData, true, null);
                        });
                    })
                </script>
